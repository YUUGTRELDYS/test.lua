local function DupeSelectedItem()
    if not MoveItem then
        notify("MoveItem remote not found")
        return
    end
    
    local inv = player.PlayerGui:FindFirstChild("HUDGui")
    if not inv then return end
    inv = inv:FindFirstChild("Inventory")
    if not inv then return end
    local slots = inv:FindFirstChild("Slots")
    if not slots then return end
    local slot1 = slots:FindFirstChild("Slot-1")
    if not slot1 then return end
    slot1 = slot1:FindFirstChild("SlotNA")
    if not slot1 then return end
    local slot1Count = slot1:FindFirstChild("Count")
    if not slot1Count then return end
    
    if slot1Count.Text == "0" then
        notify("Select item first")
        return
    end
    
    if slot1Count.Text == "64" then
        notify("Item limit reached")
        return
    end
    
    local targetSlot = nil
    for _, slotFrame in pairs(slots:GetChildren()) do
        if slotFrame.Name ~= "Slot-1" and slotFrame:FindFirstChild("Slot") then
            local display1 = slot1:FindFirstChild("Display")
            local display2 = slotFrame.Slot:FindFirstChild("Display")
            
            if display1 and display2 then
                local child1 = display1:GetChildren()[2]
                local child2 = display2:GetChildren()[2]
                
                if child1 and child2 and child1.Name == child2.Name then
                    targetSlot = slotFrame.Name
                    break
                end
            end
        end
    end
    
    if not targetSlot then
        local howmuch = 64 - tonumber(slot1Count.Text)
        local success, err = pcall(function()
            MoveItem:InvokeServer(-1, 82, true, -howmuch)
        end)
        
        if success then
            notify("Dupe attempted")
        else
            notify("Need 2 identical items")
        end
        return
    end
    
    local slotNum = tonumber(string.match(targetSlot, "%d+"))
    local howmuch = 64 - tonumber(slot1Count.Text)
    local success, err = pcall(function()
        MoveItem:InvokeServer(-1, slotNum, true, -howmuch)
    end)
    
    if success then
        notify("Duping successful")
    else
        notify("Failed: " .. tostring(err))
    end
end

local partName2Color = {
    CoalOre = "Really black",
    SteelOre = "Medium stone grey",
    GoldOre = "Gold",
    DiamondOre = "Cyan",
    RubyOre = "Really red",
    SapphireOre = "Really blue",
    Wheat5 = "Wheat"
}

local function createESP(adornee, color)
    local a = Instance.new("BoxHandleAdornment")
    a.Parent = adornee
    a.Adornee = adornee
    a.AlwaysOnTop = true
    a.ZIndex = 0
    a.Size = adornee.Size
    a.Transparency = 0.5
    a.Color = BrickColor.new(color)
    return a
end

local function initESP(name, state)
    if not workspace:FindFirstChild("Blocks") then return end
    
    local blocks = workspace.Blocks
    
    if state then
        local color = partName2Color[name]
        if color == nil then
            color = 'Institutional white'
        end
        for _, v in pairs(blocks:GetDescendants()) do
            if v.Name == name and not v:FindFirstChildOfClass("BoxHandleAdornment") then
                createESP(v, color)
            end
        end
    else
        for _, v in pairs(blocks:GetDescendants()) do
            if v:IsA('BoxHandleAdornment') and v.Parent.Name == name then
                v:Destroy()
            end
        end
    end
end

local function clearAllESP()
    if not workspace:FindFirstChild("Blocks") then return end
    
    local blocks = workspace.Blocks
    for _, v in pairs(blocks:GetDescendants()) do
        if v:IsA('BoxHandleAdornment') then
            v:Destroy()
        end
    end
end

local function setupBlockListener()
    if not workspace:FindFirstChild("Blocks") then return end
    
    local blocks = workspace.Blocks
    blocks.DescendantAdded:Connect(function(v)
        if v.Name == 'CoalOre' and xrayESPStates.CoalOre then
            createESP(v, "Really black")
        elseif v.Name == 'SteelOre' and xrayESPStates.SteelOre then
            createESP(v, "Medium stone grey")
        elseif v.Name == 'GoldOre' and xrayESPStates.GoldOre then
            createESP(v, "Gold")
        elseif v.Name == 'DiamondOre' and xrayESPStates.DiamondOre then
            createESP(v, "Cyan")
        elseif v.Name == 'RubyOre' and xrayESPStates.RubyOre then
            createESP(v, "Really red")
        elseif v.Name == 'SapphireOre' and xrayESPStates.SapphireOre then
            createESP(v, "Really blue")
        elseif v.Name == 'Wheat5' and xrayESPStates.Wheat5 then
            createESP(v, "Wheat")
        elseif customBlockName ~= "" and v.Name == customBlockName and xrayESPStates.Custom then
            createESP(v, "Institutional white")
        end
    end)
end

local colorCorrection, atmosphereEffect

local function enableAtmosphere()
    lighting.FogColor = Color3.fromRGB(100, 0, 150)
    lighting.FogEnd = 500
    lighting.FogStart = 50
    lighting.Brightness = 1.2
    lighting.ClockTime = 18
    lighting.Ambient = Color3.fromRGB(100, 100, 150)

    if not colorCorrection then
        colorCorrection = Instance.new("ColorCorrectionEffect")
        colorCorrection.Name = "AtmosphereColorCorrection"
        colorCorrection.Contrast = 0.1
        colorCorrection.Saturation = 0.5
        colorCorrection.TintColor = Color3.fromRGB(150, 100, 200)
        colorCorrection.Parent = lighting
    else
        colorCorrection.Enabled = true
    end

    if not atmosphereEffect then
        atmosphereEffect = Instance.new("Atmosphere")
        atmosphereEffect.Name = "CustomAtmosphere"
        atmosphereEffect.Density = 0.3
        atmosphereEffect.Offset = 0.25
        atmosphereEffect.Color = Color3.fromRGB(150, 50, 200)
        atmosphereEffect.Decay = Color3.fromRGB(50, 150, 50)
        atmosphereEffect.Glare = 0.1
        atmosphereEffect.Haze = 2
        atmosphereEffect.Parent = lighting
    else
        atmosphereEffect.Enabled = true
    end
end

local function disableAtmosphere()
    for setting, value in pairs(originalSettings) do
        lighting[setting] = value
    end

    if colorCorrection then
        colorCorrection.Enabled = false
    end

    if atmosphereEffect then
        atmosphereEffect.Enabled = false
    end
end

    Button.MouseLeave:Connect(function()
        TweenService:Create(Button, TweenInfo.new(0.2), {BackgroundTransparency = 0.3}):Play()

    return Button
end

            toggleButton.Size = UDim2.new(0.25, 0, 0.7, 0)
            toggleButton.Position = UDim2.new(0.7, 0, 0.15, 0)
            
            local isIgnored = table.find(ignoredPlayers, otherPlayer.Name) ~= nil
            playerStates[otherPlayer.Name] = isIgnored
            
            if isIgnored then
                toggleButton.BackgroundColor3 = Color3.fromRGB(0, 150, 0)
                toggleButton.Text = "✅"
            else
                toggleButton.BackgroundColor3 = Color3.fromRGB(150, 0, 0)
                toggleButton.Text = "❌"
            end
            
            toggleButton.TextColor3 = Color3.fromRGB(255, 255, 255)
            toggleButton.Font = Enum.Font.GothamBold
            toggleButton.TextSize = 16
            toggleButton.Parent = playerFrame
            
            local toggleButtonCorner = Instance.new("UICorner")
            toggleButtonCorner.CornerRadius = UDim.new(0, 6)
            toggleButtonCorner.Parent = toggleButton
            
            toggleButton.MouseButton1Click:Connect(function()
                local currentState = playerStates[otherPlayer.Name] or false
                local newState = not currentState
                playerStates[otherPlayer.Name] = newState
                
                if newState then
                    if not table.find(ignoredPlayers, otherPlayer.Name) then
                        table.insert(ignoredPlayers, otherPlayer.Name)
                    end
                    toggleButton.BackgroundColor3 = Color3.fromRGB(0, 150, 0)
                    toggleButton.Text = "✅"
                else
                    for i, name in ipairs(ignoredPlayers) do
                        if name == otherPlayer.Name then
                            table.remove(ignoredPlayers, i)
                            break
                        end
                    end
                    toggleButton.BackgroundColor3 = Color3.fromRGB(150, 0, 0)
                    toggleButton.Text = "❌"
                end
            end)
        end
    end
    
    teamListScrolling.CanvasSize = UDim2.new(0, 0, 0, teamListLayout.AbsoluteContentSize.Y + 10)
end

local function updateSpeedSlider()
    local fillAmount = (speedMultiplier - 1) / 9
    speedFill.Size = UDim2.new(fillAmount, 0, 1, 0)
    speedValueLabel.Text = "Speed: " .. string.format("%.1f", speedMultiplier) .. "x"
end

updateSpeedSlider()

speedSlider.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        local connection
        connection = RunService.RenderStepped:Connect(function()
            local mousePos = UserInputService:GetMouseLocation()
            local sliderPos = speedSlider.AbsolutePosition
            local sliderSize = speedSlider.AbsoluteSize
            
            local relativeX = math.clamp((mousePos.X - sliderPos.X) / sliderSize.X, 0, 1)
            speedMultiplier = 1 + (relativeX * 9)
            updateSpeedSlider()
            
            if speedActive then
                local character = player.Character
                if character and character:FindFirstChild("Humanoid") then
                    character.Humanoid.WalkSpeed = originalWalkSpeed * speedMultiplier
                end
            end
        end)
        
        UserInputService.InputEnded:Connect(function(inputEnded)
            if inputEnded.UserInputType == Enum.UserInputType.MouseButton1 then
                connection:Disconnect()
            end
        end)
    end
end)

local function getClosestPlayer()
    local closestPlayer = nil
    local shortestDistance = math.huge
    local myCharacter = player.Character
    local myRoot = myCharacter and myCharacter:FindFirstChild("HumanoidRootPart")

    if not myCharacter or not myRoot then return nil end

    for _, v in pairs(Players:GetPlayers()) do
        if v ~= player and v.Character and v.Character.PrimaryPart then
            local targetRoot = v.Character:FindFirstChild("HumanoidRootPart")
            local targetHumanoid = v.Character:FindFirstChild("Humanoid")
            
            if targetRoot and targetHumanoid and targetHumanoid.Health > 0 then
                if not isPlayerIgnored(v.Name) then
                    local distance = (targetRoot.Position - myRoot.Position).magnitude
                    if distance < shortestDistance then
                        closestPlayer = v.Character
                        shortestDistance = distance
                    end
                end
            end
        end
    end

    return closestPlayer
end

local radius = 15
local numParts = 36
local circleParts = {}
local circlePartsFolder

local function createCircle()
    if circlePartsFolder then
        circlePartsFolder:Destroy()
        circlePartsFolder = nil
    end
    
    for i = #circleParts, 1, -1 do
        if circleParts[i] then
            circleParts[i]:Destroy()
            table.remove(circleParts, i)
        end
    end
    
    circlePartsFolder = Instance.new("Folder")
    circlePartsFolder.Name = "KillAuraCircle"
    
    for i = 1, numParts do
        local part = Instance.new("Part")
        part.Name = "CirclePart" .. i
        part.Anchored = true
        part.CanCollide = false
        part.Material = Enum.Material.Neon
        part.Color = Color3.fromRGB(128, 0, 255)
        part.Size = Vector3.new(1, 0.1, (2 * math.pi * radius) / numParts)
        part.Transparency = 0.3
        part.Parent = circlePartsFolder
        table.insert(circleParts, part)
    end
    
    return circlePartsFolder
end

local function updateCirclePosition()
    if not killAuraActive then
        if circlePartsFolder then
            circlePartsFolder:Destroy()
            circlePartsFolder = nil
        end
        return
    end
    
    local character = player.Character
    if not character then
        if circlePartsFolder then
            circlePartsFolder:Destroy()
            circlePartsFolder = nil
        end
        return
    end
    
    local humanoid = character:FindFirstChild("Humanoid")
    if not humanoid or humanoid.Health <= 0 then
        if circlePartsFolder then
            circlePartsFolder:Destroy()
            circlePartsFolder = nil
        end
        return
    end
    
    local rootPart = character:FindFirstChild("HumanoidRootPart")
    if not rootPart then
        if circlePartsFolder then
            circlePartsFolder:Destroy()
            circlePartsFolder = nil
        end
        return
    end
    
    if not circlePartsFolder or not circlePartsFolder.Parent then
        circlePartsFolder = createCircle()
        if circlePartsFolder then
            circlePartsFolder.Parent = workspace
        end
    end
    
    if not circlePartsFolder or #circleParts == 0 then return end
    
    local cframe = rootPart.CFrame
    local angleIncrement = math.rad(360 / numParts)
    
    for i, part in ipairs(circleParts) do
        if part and part.Parent then
            local angle = angleIncrement * i
            local x = math.cos(angle) * radius
            local z = math.sin(angle) * radius
            
            part.CFrame = cframe * CFrame.new(x, -2, z) * CFrame.Angles(0, -angle, 0)
            part.Color = killAuraActive and Color3.fromRGB(128, 0, 255) or Color3.fromRGB(255, 0, 0)
            part.Transparency = killAuraActive and 0.3 or 0.5
        end
    end
end

local function cleanupCircle()
    if circlePartsFolder then
        circlePartsFolder:Destroy()
        circlePartsFolder = nil
    end
    
    for i = #circleParts, 1, -1 do
        if circleParts[i] then
            circleParts[i]:Destroy()
        end
        circleParts[i] = nil
    end
end

            killAuraActive = not killAuraActive

            auraButton.BackgroundColor3 = killAuraActive and Color3.fromRGB(0, 100, 0) or Color3.fromRGB(100, 0, 0)

        holdStartTimeG = nil
    end
end)
